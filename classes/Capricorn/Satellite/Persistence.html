      def dump(io=nil)
        data = {}
        
        private_vars = %w( basedomain subdomain )
        instance_variables.each do |ivar_name|
          ivar_name = ivar_name.to_s
          ivar_name =~ /^@(.+)$/
          name = $1
          unless private_vars.include? name
            data[name] = instance_variable_get(ivar_name.to_sym)
          end
        end
        
        if io
          io.write YAML.dump(data)
        else
          YAML.dump(data)
        end
      end
