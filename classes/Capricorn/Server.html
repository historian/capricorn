[show source] 
# File lib/capricorn/server.rb, line 95
    def queued_jobs
      queued_jobs = []
      Capricorn.system.queue.each do |job, canceled, immediated|
        job_delay = job.delay
        queued_jobs.push([job.id, job.name, canceled, immediated, job.running?, job.waiting?, job_delay])
      end
      queued_jobs
    end
