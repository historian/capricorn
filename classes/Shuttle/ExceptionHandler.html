    def self.setup(out=STDOUT, err=STDERR)
      if String === out
        @out = Logger.new(out, 'daily')
      else
        @out = Logger.new(out)
      end
      if String === err
        @err = Logger.new(err, 'daily')
      else
        @err = Logger.new(err)
      end
      @out.level = Logger::DEBUG
      @err.level = Logger::DEBUG
    end
