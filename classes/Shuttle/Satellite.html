    def initialize(domain)
      if Hash === domain
        domain.each do |name, value|
          instance_variable_set("@#{name}".to_sym, value)
        end
      else
        @domain = domain
        @engines = {}
      end
      @domain.gsub! /^www\./, ''
    end

    def basedomain
      unless @basedomain
        parts = self.domain.split('.')

        @basedomain = parts.join('.')
      end
